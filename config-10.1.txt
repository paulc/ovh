#!/bin/sh

# Remove RTM
sed -i '' '/rtm/d' /etc/crontab
rmuser -vy ovh
rm -rf /usr/local/rtm/
pkg delete -f \*

# Initialise pkg
ASSUME_ALWAYS_YES=1 pkg bootstrap -f

# Disable sendmail
sysrc sendmail_enable=NONE

# Change shell to /bin/sh
chsh -s /bin/sh

# Set password
passwd

# Update system
freebsd-update fetch
freebsd-update install

# Reboot
reboot
